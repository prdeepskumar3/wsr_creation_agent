```python
"""
Risk Schema - Part of WSR Input Schema
This is NOT a separate module. Risks are part of WSR data entry.
"""

from typing import Optional
from datetime import date
from pydantic import BaseModel, Field, validator
from enum import Enum


class RiskStatus(str, Enum):
    OPEN = "Open"
    IN_PROGRESS = "In-Progress"
    CLOSED = "Closed"


class RiskSeverity(str, Enum):
    HIGH = "High"
    MEDIUM = "Medium"
    LOW = "Low"


class RiskSchema(BaseModel):
    """Risk schema - part of WSR input"""
    
    id: Optional[str] = None
    risk_id: Optional[str] = None  # Auto-generated sequence number
    description: str = Field(..., min_length=10, max_length=500)
    status: RiskStatus
    severity: RiskSeverity
    owner: str = Field(..., min_length=2, max_length=100)
    mitigation: str = Field(..., min_length=10, max_length=1000)
    planned_closure_date: date
    remark: Optional[str] = Field(None, max_length=500)
    
    @validator('planned_closure_date')
    def validate_future_date(cls, v):
        if v < date.today():
            raise ValueError('Planned closure date must be in the future')
        return v
    
    @validator('remark')
    def validate_remark_for_closed(cls, v, values):
        """Remark is mandatory when status is Closed"""
        if values.get('status') == RiskStatus.CLOSED:
            if not v or len(v.strip()) < 5:
                raise ValueError('Remark is mandatory when risk status is Closed')
        return v
    
    @validator('mitigation')
    def validate_mitigation_for_high(cls, v, values):
        """High severity risks need detailed mitigation"""
        if values.get('severity') == RiskSeverity.HIGH:
            if len(v) < 15:
                raise ValueError('High severity risks require detailed mitigation (min 15 chars)')
        return v
```

```python
"""
Risk Rules Module - Business Rules for Risk Validation
This is called from validate_node during WSR validation
"""

from typing import List, Optional
from datetime import date
from dataclasses import dataclass


@dataclass
class ValidationResult:
    """Validation result container"""
    is_valid: bool
    errors: List[str]


class RiskRules:
    """Business rules for risk validation - called from WSR validation"""
    
    RISK_STATUSES = ["Open", "In-Progress", "Closed"]
    RISK_SEVERITIES = ["High", "Medium", "Low"]

    @classmethod
    def validate_risk(cls, risk_data: dict) -> ValidationResult:
        """Validate a single risk entry (called during WSR validation)"""
        errors = []

        # 1. Required fields
        required_fields = [
            "description", "status", "severity",
            "owner", "mitigation", "planned_closure_date"
        ]
        for field in required_fields:
            if not risk_data.get(field):
                errors.append(f"{field} is required")

        # 2. Field length validations
        if risk_data.get("description"):
            if len(risk_data["description"]) < 10:
                errors.append("Description must be at least 10 characters")

        if risk_data.get("owner"):
            if len(risk_data["owner"]) < 2:
                errors.append("Owner must be at least 2 characters")

        if risk_data.get("mitigation"):
            if len(risk_data["mitigation"]) < 10:
                errors.append("Mitigation must be at least 10 characters")

        # 3. Status validation
        status = risk_data.get("status")
        if status and status not in cls.RISK_STATUSES:
            errors.append(f"Invalid status: {status}")

        # 4. Severity validation
        severity = risk_data.get("severity")
        if severity and severity not in cls.RISK_SEVERITIES:
            errors.append(f"Invalid severity: {severity}")

        # 5. Future date validation
        closure_date = risk_data.get("planned_closure_date")
        if closure_date:
            if closure_date < date.today():
                errors.append("Planned closure date must be in the future")

        # 6. Remark validation for Closed status
        if status == "Closed":
            remark = risk_data.get("remark")
            if not remark or len(remark.strip()) < 5:
                errors.append("Remark is mandatory when risk is Closed (minimum 5 characters)")

        # 7. High severity validation
        if severity == "High" and status != "Closed":
            if not risk_data.get("owner"):
                errors.append("High severity risk requires an owner")
            if not risk_data.get("mitigation"):
                errors.append("High severity risk requires mitigation")
            if len(risk_data.get("mitigation", "")) < 15:
                errors.append("High severity risks require detailed mitigation (min 15 characters)")
            if not risk_data.get("planned_closure_date"):
                errors.append("High severity risk requires a planned closure date")

        return ValidationResult(
            is_valid=len(errors) == 0,
            errors=errors
        )

    @classmethod
    def validate_all_risks(cls, risks: List[dict]) -> ValidationResult:
        """Validate all risks in the WSR (called during WSR validation)"""
        all_errors = []
        
        # 1. Validate each risk
        for idx, risk in enumerate(risks):
            result = cls.validate_risk(risk)
            if not result.is_valid:
                all_errors.extend([f"Risk {idx+1}: {err}" for err in result.errors])
        
        # 2. Check for duplicate descriptions
        descriptions = []
        for risk in risks:
            desc = risk.get("description", "").lower().strip()
            if desc:
                descriptions.append(desc)
        
        duplicates = [d for d in set(descriptions) if descriptions.count(d) > 1]
        for dup in duplicates:
            all_errors.append(f"Duplicate risk description: {dup}")
        
        return ValidationResult(
            is_valid=len(all_errors) == 0,
            errors=all_errors
        )

    @classmethod
    def validate_status_transition(cls, old_status: str, new_status: str) -> ValidationResult:
        """Validate risk status transitions (called when editing a risk)"""
        errors = []

        if old_status == "Open" and new_status == "Closed":
            errors.append("Cannot move from Open to Closed. Must go through In-Progress")

        if old_status == "Closed" and new_status != "Closed":
            errors.append("Cannot reopen a Closed risk. Create a new risk entry.")

        return ValidationResult(
            is_valid=len(errors) == 0,
            errors=errors
        )
```

```python
"""
WSR Service Module - Business Logic for Weekly Status Reports
"""

from typing import List, Optional
from datetime import datetime
from fastapi import HTTPException

from src.repositories.wsr_repository import WSRRepository
from src.services.insight_service import InsightService
from src.services.audit_service import AuditService
from src.domain.validation_rules import validate_wsr_input
from src.domain.risk_rules import RiskRules
from src.schemas.wsr import WSRInputSchema, WSRResponse
from src.models.wsr import WSRStatus, ApprovalStatus


class WSRService:
    """Service for managing Weekly Status Reports"""

    def __init__(
        self,
        repository: WSRRepository,
        insight_service: InsightService,
        audit_service: AuditService
    ):
        self.repository = repository
        self.insight_service = insight_service
        self.audit_service = audit_service

    async def create_wsr(self, data: WSRInputSchema, user_id: str) -> WSRResponse:
        """Create a new WSR with validation"""
        
        # 1. Validate all WSR data (including risks)
        validation_result = validate_wsr_input(data.dict())
        if not validation_result.is_valid:
            raise HTTPException(
                status_code=400,
                detail=validation_result.errors
            )
        
        # 2. Validate risks specifically
        risk_result = RiskRules.validate_all_risks(data.risks)
        if not risk_result.is_valid:
            raise HTTPException(
                status_code=400,
                detail=risk_result.errors
            )

        # 3. Create WSR
        wsr = await self.repository.create(data.dict(), user_id)

        # 4. Log audit
        await self.audit_service.log_event(
            user_id=user_id,
            event_type="wsr_create",
            action="create",
            after_data=wsr.dict()
        )

        return wsr

    async def update_wsr(self, wsr_id: str, data: WSRInputSchema, user_id: str) -> WSRResponse:
        """Update existing WSR (including risks)"""
        
        # 1. Get current state
        current = await self.repository.get_by_id(wsr_id)
        if not current:
            raise HTTPException(status_code=404, detail="WSR not found")

        # 2. Validate updates
        validation_result = validate_wsr_input(data.dict())
        if not validation_result.is_valid:
            raise HTTPException(
                status_code=400,
                detail=validation_result.errors
            )
        
        # 3. Validate risks
        risk_result = RiskRules.validate_all_risks(data.risks)
        if not risk_result.is_valid:
            raise HTTPException(
                status_code=400,
                detail=risk_result.errors
            )

        # 4. Log before state
        before_data = current.dict()

        # 5. Update
        updated = await self.repository.update(wsr_id, data.dict(), user_id)

        # 6. Log after state
        await self.audit_service.log_event(
            user_id=user_id,
            event_type="wsr_update",
            action="update",
            before_data=before_data,
            after_data=updated.dict()
        )

        return updated

    async def generate_wsr(self, wsr_id: str, user_id: str) -> dict:
        """Generate AI-enhanced WSR"""
        
        # 1. Get WSR data
        wsr = await self.repository.get_by_id(wsr_id)
        if not wsr:
            raise HTTPException(status_code=404, detail="WSR not found")

        # 2. Trigger workflow
        from src.workflow.graph import run_wsr_workflow
        result = await run_wsr_workflow(wsr.dict(), user_id)

        # 3. Generate insights
        insights = await self.insight_service.generate_insights(result)

        # 4. Update WSR with generated content
        await self.repository.update_generated_content(
            wsr_id,
            result["content"],
            insights
        )

        return result
```

```python
"""
Validate Node - Validates all WSR input including risks
"""

from typing import Dict, Any
from langgraph.types import Command

from src.domain.validation_rules import validate_wsr_input
from src.domain.risk_rules import RiskRules


def validate_node(state: Dict[str, Any]) -> Command:
    """
    Validate all WSR input data including risks.
    This is the first step in the workflow.
    """
    
    # 1. Validate WSR input (all fields)
    wsr_data = state.get("wsr_input", {})
    validation_result = validate_wsr_input(wsr_data)
    
    if not validation_result.is_valid:
        return Command(
            goto="error",
            update={
                "error": True,
                "error_messages": validation_result.errors
            }
        )
    
    # 2. Validate risks (part of WSR input)
    risks = wsr_data.get("risks", [])
    risk_result = RiskRules.validate_all_risks(risks)
    
    if not risk_result.is_valid:
        return Command(
            goto="error",
            update={
                "error": True,
                "error_messages": risk_result.errors
            }
        )
    
    # 3. All validations passed
    return Command(
        goto="route_model",
        update={
            "validated_data": wsr_data,
            "validation_passed": True,
            "risk_data": risks  # Pass risk data for analysis
        }
    )
```

## 4.2 Risk Section Component (Part of WSR Form)

```tsx
/**
 * RiskSection Component - Part of WSR Form
 * PM enters risks as part of the WSR input
 */

import React, { useState } from 'react';
import { useWsrStore } from '@/stores/wsrStore';
import { RiskTable } from '@/components/risks/RiskTable';
import { RiskModal } from '@/components/risks/RiskModal';
import { Button } from '@/components/common/Button';
import { useRiskValidation } from '../hooks/useRiskValidation';

interface RiskSectionProps {
  isReadOnly?: boolean;
}

export const RiskSection: React.FC<RiskSectionProps> = ({ isReadOnly = false }) => {
  const { risks, addRisk, updateRisk, deleteRisk } = useWsrStore();
  const { validateRisk, errors } = useRiskValidation();
  const [isModalOpen, setIsModalOpen] = useState(false);
  const [editingRisk, setEditingRisk] = useState(null);

  const handleAddRisk = () => {
    setEditingRisk(null);
    setIsModalOpen(true);
  };

  const handleEditRisk = (risk) => {
    setEditingRisk(risk);
    setIsModalOpen(true);
  };

  const handleSaveRisk = (riskData) => {
    // Validate risk before saving
    const validation = validateRisk(riskData);
    if (!validation.isValid) {
      // Show validation errors
      return;
    }

    if (editingRisk) {
      updateRisk(editingRisk.id, riskData);
    } else {
      addRisk(riskData);
    }
    setIsModalOpen(false);
  };

  const handleDeleteRisk = (riskId) => {
    if (confirm('Are you sure you want to delete this risk?')) {
      deleteRisk(riskId);
    }
  };

  return (
    <div className="risk-section">
      <div className="section-header">
        <h3>Risks & Dependencies</h3>
        {!isReadOnly && (
          <Button onClick={handleAddRisk}>
            <i className="fas fa-plus" /> Add Risk
          </Button>
        )}
      </div>

      <RiskTable
        risks={risks}
        onEdit={handleEditRisk}
        onDelete={handleDeleteRisk}
        isReadOnly={isReadOnly}
      />

      {errors.length > 0 && (
        <div className="validation-errors">
          {errors.map((error, index) => (
            <div key={index} className="error-message">
              {error}
            </div>
          ))}
        </div>
      )}

      <div className="risk-rules-hint">
        <p>Note: When closing a risk, remarks are mandatory.</p>
      </div>

      <RiskModal
        isOpen={isModalOpen}
        onClose={() => setIsModalOpen(false)}
        onSave={handleSaveRisk}
        initialData={editingRisk}
      />
    </div>
  );
};
```

```sql
-- ============================================================
-- ACCOUNTS TABLE
-- ============================================================
CREATE TABLE accounts (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name VARCHAR(255) NOT NULL,
    description TEXT,
    settings JSONB DEFAULT '{}',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- ============================================================
-- PROJECTS TABLE
-- ============================================================
CREATE TABLE projects (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    account_id UUID NOT NULL REFERENCES accounts(id) ON DELETE CASCADE,
    name VARCHAR(255) NOT NULL,
    description TEXT,
    delivery_model VARCHAR(50) NOT NULL,
    settings JSONB DEFAULT '{}',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- ============================================================
-- WSR REPORTS TABLE (Risk data stored as JSON)
-- ============================================================
CREATE TABLE wsr_reports (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    account_id UUID NOT NULL REFERENCES accounts(id) ON DELETE CASCADE,
    project_id UUID NOT NULL REFERENCES projects(id) ON DELETE CASCADE,
    reporting_week DATE NOT NULL,
    status VARCHAR(20) NOT NULL DEFAULT 'draft',
    tracking_model VARCHAR(50) NOT NULL,
    model_payload JSONB NOT NULL DEFAULT '{}',
    model_metrics JSONB NOT NULL DEFAULT '{}',
    model_findings JSONB NOT NULL DEFAULT '[]',
    overall_status VARCHAR(10) NOT NULL,
    release_milestone VARCHAR(255),
    pm_highlights TEXT,
    pm_next_plan TEXT,
    pm_notes TEXT,
    risks JSONB DEFAULT '[]',  -- Risk data stored as JSON array
    ai_executive_summary TEXT,
    ai_risk_analysis TEXT,
    ai_next_week TEXT,
    ai_qa_score DECIMAL(3,2),
    ai_qa_feedback TEXT,
    approval_status VARCHAR(20) DEFAULT 'pending',
    approved_by UUID,
    approved_at TIMESTAMP,
    exported_at TIMESTAMP,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
    CONSTRAINT check_overall_status CHECK (overall_status IN ('Green', 'Amber', 'Red')),
    CONSTRAINT check_approval_status CHECK (approval_status IN ('pending', 'approved', 'rejected')),
    CONSTRAINT check_status CHECK (status IN ('draft', 'generated', 'reviewed', 'approved', 'exported'))
);

-- ============================================================
-- AI INSIGHTS TABLE
-- ============================================================
CREATE TABLE ai_insights (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    wsr_id UUID NOT NULL REFERENCES wsr_reports(id) ON DELETE CASCADE,
    insight_type VARCHAR(50) NOT NULL,
    severity VARCHAR(20) NOT NULL,
    description TEXT NOT NULL,
    evidence TEXT,
    recommendation TEXT NOT NULL,
    pm_action VARCHAR(20),
    pm_reason TEXT,
    modified_content TEXT,
    accepted BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- ============================================================
-- AUDIT LOGS TABLE
-- ============================================================
CREATE TABLE audit_logs (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    wsr_id UUID NOT NULL REFERENCES wsr_reports(id) ON DELETE CASCADE,
    user_id UUID NOT NULL,
    event_type VARCHAR(50) NOT NULL,
    action VARCHAR(50) NOT NULL,
    before_data JSONB,
    after_data JSONB,
    metadata JSONB,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- ============================================================
-- APPROVAL EVENTS TABLE
-- ============================================================
CREATE TABLE approval_events (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    wsr_id UUID NOT NULL REFERENCES wsr_reports(id) ON DELETE CASCADE,
    user_id UUID NOT NULL,
    action VARCHAR(20) NOT NULL,
    comments TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- ============================================================
-- EXPORT ATTEMPTS TABLE
-- ============================================================
CREATE TABLE export_attempts (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    wsr_id UUID NOT NULL REFERENCES wsr_reports(id) ON DELETE CASCADE,
    format VARCHAR(10) NOT NULL,
    status VARCHAR(20) NOT NULL,
    file_path VARCHAR(500),
    error_message TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    completed_at TIMESTAMP
);

-- ============================================================
-- VALIDATION FUNCTIONS FOR RISK DATA
-- ============================================================
CREATE OR REPLACE FUNCTION validate_risk_json(risk_data JSONB)
RETURNS BOOLEAN AS $$
BEGIN
    -- Check required fields
    IF NOT (risk_data ? 'description' AND risk_data ? 'status' AND risk_data ? 'severity') THEN
        RETURN FALSE;
    END IF;
    
    -- Check status values
    IF NOT (risk_data->>'status' IN ('Open', 'In-Progress', 'Closed')) THEN
        RETURN FALSE;
    END IF;
    
    -- Check severity values
    IF NOT (risk_data->>'severity' IN ('High', 'Medium', 'Low')) THEN
        RETURN FALSE;
    END IF;
    
    -- Check remark on closed status
    IF risk_data->>'status' = 'Closed' THEN
        IF NOT (risk_data ? 'remark' AND char_length(risk_data->>'remark') >= 5) THEN
            RETURN FALSE;
        END IF;
    END IF;
    
    RETURN TRUE;
END;
$$ LANGUAGE plpgsql;
```

# 8. Summary

## 8.1 Key Design Principles

| Principle | Implementation |
|-----------|----------------|
| **Risk is NOT a separate module** | Risk data is part of WSR input schema |
| **Single WSR Service** | All operations handled through WSR service |
| **Validation at entry** | Risks validated during WSR validation |
| **JSON storage** | Risks stored as JSON in wsr_reports table |
| **No separate risk tables** | Simplified data model |

## 8.2 Module Dependencies

```
api/ вҶ’ services/
services/ вҶ’ repositories/, domain/
workflow/ вҶ’ services/, agents/
domain/ вҶ’ None (pure business logic)
repositories/ вҶ’ models/
```
